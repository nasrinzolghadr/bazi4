# بازی کلیک روی سکه - PWA آماده APK

این پوشه آماده آپلود روی GitHub Pages است.

فایل‌ها:
- index.html
- style.css
- script.js
- manifest.json
- service-worker.js
- icons/coin.png
- icons/icon-192.png
- icons/icon-512.png
